** Display float time
:display-time-world
** Display buffered time
(defun open-clock-buffer ()
  "Open a buffer displaying the current time."
  (interactive)
  (let ((buf (get-buffer-create "*Clock*")))
    (with-current-buffer buf
      (erase-buffer)
      (insert (format-time-string "%A, %d %B %Y\n%H:%M:%S"))
      (goto-char (point-min))
      (display-buffer buf)
      (run-with-timer 1 nil (lambda ()
                              (with-current-buffer "*Clock*"
                                (erase-buffer)
                                (insert (format-time-string "%A, %d %B %Y\n%H:%M:%S"))
                                (goto-char (point-min))))))))
