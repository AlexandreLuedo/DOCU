Arch User Repository
* Installing yay
#+begin_src bash
$ sudo pacma -S --needed git base-devel
$ git clone https://aur.archlinux.org/yay.git
$ cd yay
$ makepkg -si
#+end_src
* Using yay
** Update
#+begin_src bash
$ yay
# or
$ yay -Syu
#+end_src
** Installing a package
#+begin_src bash
$ yay -S package_name
#+end_src
** Searching a package
#+begin_src bash
$ yay search_term
#+end_src
** Cleaning up
*** Removed unneeded depedencies
#+begin_src bash
$ yay -Yc
#+end_src
*** Clean package cache
#+begin_src bash
$ yay -Scc
#+end_src
