man lpr

#+end_src
* Print
See also : [[http://localhost:631/help/options.html?TOPIC=Getting+Started&QUERY=][CUPS printing docs]]
#+begin_src bash
lpr /path/to/file
cat [file] | lpr
#+end_src
** Print black & white
#+begin_src bash
lpr -o color-mode=monochrome [file]
#+end_src
** Print specified pages
#+begin_src bash
lpr -o page-ranges=1 [file]
lpr -o page-ranges=1-3 [file]
#+end_src
** Print on both sides
#+begin_src bash
lp -o sides=two-sided-short-edge filename
lp -o sides=two-sided-long-edge filename
lpr -o sides=two-sided-long-edge filename
#+end_src
* View printing queue
#+begin_src bash
lpq
#+end_src
* Cancel printing job
#+begin_src bash
lprm -
lprm [jobID]
lprm -a all
cancel [printerName]
#+end_src
* Managing printer
** Setup lpr
#+begin_src bash
sudo lpadmin -d [printerName]
#+end_src
** Setting the default printer
#+begin_src bash
lpoptions -d printer
#+end_src
** Delete printer
#+begin_src bash
lpadmin -x [printerName]
#+end_src
