\label{sec:orgd0977c7}

\subsection*{Display buffered time}
\label{sec:orge5c1c91}
(defun open-clock-buffer ()
  ``Open a buffer displaying the current time.''
  (interactive)
  (let ((buf (get-buffer-create ``\textbf{Clock}'')))
    (with-current-buffer buf
      (erase-buffer)
      (insert (format-time-string ``\%A, \%d \%B \%Y\n\%H:\%M:\%S''))
      (goto-char (point-min))
      (display-buffer buf)
      (run-with-timer 1 nil (lambda ()
                              (with-current-buffer ``\textbf{Clock}''
                                (erase-buffer)
                                (insert (format-time-string ``\%A, \%d \%B \%Y\n\%H:\%M:\%S''))
                                (goto-char (point-min))))))))
